## 1019  
```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main(void) {
	
	int year = 0;
	int month = 0;
	int day = 0;

	scanf("%d.%d.%d", &year, &month, &day);
	printf("%04d.%02d.%02d", year, month, day);

	return 0;
}

```
